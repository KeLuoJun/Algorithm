## 🌳 树状数组（Fenwick Tree）

### 基本概念

树状数组（Binary Indexed Tree，BIT）是一种支持单点修改和区间查询的数据结构，时间复杂度为 O(log n)。

### 核心思想
- 利用二进制的性质，将数组分解为若干个区间
- 每个节点存储一个区间的和
- 通过 `lowbit` 操作快速定位父子节点

### 基本操作

#### 单点增加、范围查询模板

```python
class FenwickTree:
    """
    树状数组 - 单点增加、范围查询
    支持操作：
    1. update(i, delta): 将位置i的值增加delta
    2. query(i): 查询前缀和[1...i]
    3. range_query(l, r): 查询区间和[l...r]
    """
    
    def __init__(self, n):
        """
        初始化树状数组
        Args:
            n: 数组长度
        """
        self.n = n
        self.tree = [0] * (n + 1)  # 1-indexed
    
    def lowbit(self, x):
        """
        获取x的最低位1及其后面的0组成的数
        例如：lowbit(12) = lowbit(1100) = 4 (100)
        """
        return x & (-x)
    
    def update(self, i, delta):
        """
        单点更新：将位置i的值增加delta
        时间复杂度：O(log n)
        """
        while i <= self.n:
            self.tree[i] += delta
            i += self.lowbit(i)
    
    def query(self, i):
        """
        前缀和查询：查询[1...i]的和
        时间复杂度：O(log n)
        """
        result = 0
        while i > 0:
            result += self.tree[i]
            i -= self.lowbit(i)
        return result
    
    def range_query(self, l, r):
        """
        区间查询：查询[l...r]的和
        时间复杂度：O(log n)
        """
        return self.query(r) - self.query(l - 1)

# 使用示例
def test_fenwick_tree():
    """测试树状数组的基本功能"""
    # 创建长度为5的树状数组
    ft = FenwickTree(5)
    
    # 初始化数组 [1, 2, 3, 4, 5]
    for i in range(1, 6):
        ft.update(i, i)
    
    print(f"前缀和[1...3]: {ft.query(3)}")  # 输出: 6 (1+2+3)
    print(f"区间和[2...4]: {ft.range_query(2, 4)}")  # 输出: 9 (2+3+4)
    
    # 更新位置2的值，增加5
    ft.update(2, 5)
    print(f"更新后区间和[2...4]: {ft.range_query(2, 4)}")  # 输出: 14 (7+3+4)
```

#### 范围增加、单点查询模板

```python
class FenwickTreeRangeUpdate:
    """
    树状数组 - 范围增加、单点查询
    使用差分数组的思想
    """
    
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (n + 1)
    
    def lowbit(self, x):
        return x & (-x)
    
    def update(self, i, delta):
        """更新差分数组"""
        while i <= self.n:
            self.tree[i] += delta
            i += self.lowbit(i)
    
    def range_update(self, l, r, delta):
        """
        区间更新：将[l...r]区间的所有值增加delta
        时间复杂度：O(log n)
        """
        self.update(l, delta)
        self.update(r + 1, -delta)
    
    def query(self, i):
        """
        单点查询：查询位置i的值
        时间复杂度：O(log n)
        """
        result = 0
        while i > 0:
            result += self.tree[i]
            i -= self.lowbit(i)
        return result

# 使用示例
def test_range_update():
    """测试区间更新功能"""
    ft = FenwickTreeRangeUpdate(5)
    
    # 区间[1,3]增加2
    ft.range_update(1, 3, 2)
    
    # 区间[2,4]增加3
    ft.range_update(2, 4, 3)
    
    for i in range(1, 6):
        print(f"位置{i}的值: {ft.query(i)}")
    # 输出: 位置1:2, 位置2:5, 位置3:5, 位置4:3, 位置5:0
```

### 经典应用：HH的项链

#### 问题描述
HH有一串由各种漂亮的贝壳组成的项链。HH相信不同的贝壳会带来好运，所以每次散步完后，他都会随意取出一段贝壳，询问这一段中有多少不同的贝壳。

#### 算法思路
- 使用树状数组维护每种贝壳的出现次数
- 对于每个查询区间，统计不同贝壳的数量
- 利用离线处理和扫描线算法

#### 代码实现

```python
def solve_necklace(n, beads, queries):
    """
    HH的项链问题解法
    Args:
        n: 项链长度
        beads: 贝壳数组
        queries: 查询列表，每个查询为(l, r)
    """
    # 离散化贝壳编号
    unique_beads = list(set(beads))
    bead_map = {bead: i + 1 for i, bead in enumerate(unique_beads)}
    
    # 转换贝壳数组
    mapped_beads = [bead_map[bead] for bead in beads]
    
    # 树状数组
    ft = FenwickTree(len(unique_beads))
    
    # 处理查询（离线算法）
    query_with_index = [(queries[i][0], queries[i][1], i) for i in range(len(queries))]
    query_with_index.sort(key=lambda x: x[1])  # 按右端点排序
    
    results = [0] * len(queries)
    last_pos = {}  # 记录每种贝壳最后出现的位置
    j = 0
    
    for l, r, idx in query_with_index:
        # 处理到位置r
        while j < r:
            bead = mapped_beads[j]
            
            # 如果这种贝壳之前出现过，先移除之前的贡献
            if bead in last_pos:
                ft.update(last_pos[bead], -1)
            
            # 在当前位置添加贡献
            ft.update(j + 1, 1)
            last_pos[bead] = j + 1
            j += 1
        
        # 查询区间[l, r]中不同贝壳的数量
        results[idx] = ft.range_query(l, r)
    
    return results
```

## 📚 并查集（Disjoint Set Union）

### 基本概念

并查集是一种树型的数据结构，用于处理一些不交集的合并及查询问题。它支持两种操作：
1. **Find**: 确定元素属于哪一个子集
2. **Union**: 将两个子集合并成同一个集合

### 核心优化
- **路径压缩**: 在查找过程中，将路径上的所有节点直接连接到根节点
- **按秩合并**: 总是将较小的树连接到较大的树上

### 代码实现

```python
class UnionFind:
    """
    并查集数据结构
    支持路径压缩和按秩合并优化
    """
    
    def __init__(self, n):
        """
        初始化并查集
        Args:
            n: 元素个数
        """
        self.parent = list(range(n))  # 父节点数组
        self.rank = [0] * n          # 秩数组（树的高度）
        self.count = n               # 连通分量个数
    
    def find(self, x):
        """
        查找元素x的根节点（带路径压缩）
        时间复杂度：O(α(n))，α为阿克曼函数的反函数
        """
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])  # 路径压缩
        return self.parent[x]
    
    def union(self, x, y):
        """
        合并元素x和y所在的集合（按秩合并）
        时间复杂度：O(α(n))
        """
        root_x = self.find(x)
        root_y = self.find(y)
        
        if root_x == root_y:
            return False  # 已经在同一个集合中
        
        # 按秩合并：将秩小的树连接到秩大的树上
        if self.rank[root_x] < self.rank[root_y]:
            self.parent[root_x] = root_y
        elif self.rank[root_x] > self.rank[root_y]:
            self.parent[root_y] = root_x
        else:
            self.parent[root_y] = root_x
            self.rank[root_x] += 1
        
        self.count -= 1
        return True
    
    def connected(self, x, y):
        """
        判断元素x和y是否在同一个集合中
        """
        return self.find(x) == self.find(y)
    
    def get_count(self):
        """
        获取连通分量的个数
        """
        return self.count

# 使用示例
def test_union_find():
    """测试并查集的基本功能"""
    uf = UnionFind(5)
    
    print(f"初始连通分量数: {uf.get_count()}")  # 5
    
    # 合并操作
    uf.union(0, 1)
    uf.union(2, 3)
    print(f"合并后连通分量数: {uf.get_count()}")  # 3
    
    # 连通性查询
    print(f"0和1是否连通: {uf.connected(0, 1)}")  # True
    print(f"0和2是否连通: {uf.connected(0, 2)}")  # False
    
    # 继续合并
    uf.union(1, 2)
    print(f"最终连通分量数: {uf.get_count()}")  # 2
```

### 经典应用：找出知晓秘密的所有专家

#### 问题描述
给你一个整数 n，表示有 n 个专家从 0 到 n - 1 编号。另外给你一个下标从 0 开始的二维整数数组 meetings，其中 meetings[i] = [xi, yi, timei] 表示专家 xi 和专家 yi 在时间 timei 要开一场会。一个专家可以同时参加多场会议。最初，专家 0 知道这个秘密。

#### 算法思路
- 按时间排序所有会议
- 对于同一时间的会议，使用并查集合并参与者
- 检查每个时间点后，哪些专家与专家0在同一个连通分量中

#### 代码实现

```python
def findAllPeople(n: int, meetings: List[List[int]], firstPerson: int) -> List[int]:
    """
    找出知晓秘密的所有专家
    """
    # 按时间排序会议
    meetings.sort(key=lambda x: x[2])
    
    # 初始化并查集
    uf = UnionFind(n)
    uf.union(0, firstPerson)  # 专家0和firstPerson最初知道秘密
    
    i = 0
    while i < len(meetings):
        current_time = meetings[i][2]
        people_in_meetings = set()
        
        # 处理同一时间的所有会议
        while i < len(meetings) and meetings[i][2] == current_time:
            x, y, _ = meetings[i]
            uf.union(x, y)
            people_in_meetings.add(x)
            people_in_meetings.add(y)
            i += 1
        
        # 检查这些人是否与专家0连通，如果不连通则重置
        for person in people_in_meetings:
            if not uf.connected(0, person):
                uf.parent[person] = person
                uf.rank[person] = 0
    
    # 找出所有与专家0连通的专家
    result = []
    for i in range(n):
        if uf.connected(0, i):
            result.append(i)
    
    return result

# 优化版本：使用临时并查集
def findAllPeopleOptimized(n: int, meetings: List[List[int]], firstPerson: int) -> List[int]:
    """
    优化版本：对每个时间点使用独立的并查集
    """
    # 按时间分组会议
    time_to_meetings = {}
    for x, y, time in meetings:
        if time not in time_to_meetings:
            time_to_meetings[time] = []
        time_to_meetings[time].append((x, y))
    
    # 知道秘密的人
    know_secret = {0, firstPerson}
    
    # 按时间顺序处理
    for time in sorted(time_to_meetings.keys()):
        # 为当前时间创建临时并查集
        people = set()
        for x, y in time_to_meetings[time]:
            people.add(x)
            people.add(y)
        
        people_list = list(people)
        people_to_idx = {p: i for i, p in enumerate(people_list)}
        
        temp_uf = UnionFind(len(people_list))
        
        # 合并会议参与者
        for x, y in time_to_meetings[time]:
            temp_uf.union(people_to_idx[x], people_to_idx[y])
        
        # 检查哪些人能获得秘密
        secret_holders = set()
        for person in people_list:
            if person in know_secret:
                root = temp_uf.find(people_to_idx[person])
                for other_person in people_list:
                    if temp_uf.find(people_to_idx[other_person]) == root:
                        secret_holders.add(other_person)
        
        know_secret.update(secret_holders)
    
    return sorted(list(know_secret))
```

## 📊 单调栈（Monotonic Stack）

### 基本概念

单调栈是一种特殊的栈，栈中的元素保持单调性（单调递增或单调递减）。它常用于解决"下一个更大元素"、"最大矩形面积"等问题。

### 核心思想
- 维护栈的单调性
- 当新元素破坏单调性时，弹出栈顶元素
- 弹出过程中可以计算相关信息

### 基本模板

```python
def monotonic_stack_template(nums):
    """
    单调栈模板
    这里以单调递增栈为例（栈底到栈顶递增）
    """
    stack = []
    result = []
    
    for i, num in enumerate(nums):
        # 当栈不为空且当前元素破坏单调性时
        while stack and nums[stack[-1]] > num:
            # 弹出栈顶元素并处理
            top_idx = stack.pop()
            # 在这里可以计算相关信息
            # 例如：找到了nums[top_idx]右边第一个更小的元素nums[i]
        
        # 将当前元素入栈
        stack.append(i)
    
    return result

def next_greater_element(nums):
    """
    找到每个元素右边第一个更大的元素
    """
    n = len(nums)
    result = [-1] * n
    stack = []  # 存储索引
    
    for i in range(n):
        # 当前元素比栈顶元素大，说明找到了栈顶元素的下一个更大元素
        while stack and nums[stack[-1]] < nums[i]:
            idx = stack.pop()
            result[idx] = nums[i]
        
        stack.append(i)
    
    return result

# 测试用例
def test_next_greater():
    nums = [2, 1, 2, 4, 3, 1]
    result = next_greater_element(nums)
    print(f"输入: {nums}")
    print(f"下一个更大元素: {result}")
    # 输出: [4, 2, 4, -1, -1, -1]
```

### 经典应用：最大矩形面积

#### 问题描述
给定 n 个非负整数，用来表示柱状图中各个柱子的高度。每个柱子彼此相邻，且宽度为 1。求在该柱状图中，能够勾勒出来的矩形的最大面积。

#### 算法思路
- 对于每个柱子，找到它左边和右边第一个比它矮的柱子
- 以当前柱子的高度为矩形高度，计算最大宽度
- 使用单调栈优化查找过程

#### 代码实现

```python
def largestRectangleArea(heights):
    """
    柱状图中最大的矩形
    使用单调栈解法
    """
    stack = []
    max_area = 0
    
    # 在数组末尾添加0，确保所有元素都能被处理
    heights = heights + [0]
    
    for i, h in enumerate(heights):
        # 当前高度小于栈顶高度时，计算面积
        while stack and heights[stack[-1]] > h:
            height = heights[stack.pop()]
            
            # 计算宽度
            width = i if not stack else i - stack[-1] - 1
            
            # 更新最大面积
            max_area = max(max_area, height * width)
        
        stack.append(i)
    
    return max_area

def largestRectangleAreaOptimized(heights):
    """
    优化版本：预处理左右边界
    """
    n = len(heights)
    
    # 找到每个位置左边第一个更小的元素
    left = [-1] * n
    stack = []
    for i in range(n):
        while stack and heights[stack[-1]] >= heights[i]:
            stack.pop()
        if stack:
            left[i] = stack[-1]
        stack.append(i)
    
    # 找到每个位置右边第一个更小的元素
    right = [n] * n
    stack = []
    for i in range(n-1, -1, -1):
        while stack and heights[stack[-1]] >= heights[i]:
            stack.pop()
        if stack:
            right[i] = stack[-1]
        stack.append(i)
    
    # 计算最大面积
    max_area = 0
    for i in range(n):
        width = right[i] - left[i] - 1
        area = heights[i] * width
        max_area = max(max_area, area)
    
    return max_area

# 测试用例
def test_largest_rectangle():
    heights = [2, 1, 5, 6, 2, 3]
    result = largestRectangleArea(heights)
    print(f"柱状图高度: {heights}")
    print(f"最大矩形面积: {result}")  # 输出: 10
```

#### 问题：每日温度

```python
def dailyTemperatures(temperatures):
    """
    每日温度问题
    找到每天之后第一个更暖和的天数
    """
    n = len(temperatures)
    result = [0] * n
    stack = []  # 存储索引
    
    for i in range(n):
        # 当前温度比栈顶温度高
        while stack and temperatures[stack[-1]] < temperatures[i]:
            prev_day = stack.pop()
            result[prev_day] = i - prev_day
        
        stack.append(i)
    
    return result

# 测试
def test_daily_temperatures():
    temps = [73, 74, 75, 71, 69, 72, 76, 73]
    result = dailyTemperatures(temps)
    print(f"温度: {temps}")
    print(f"等待天数: {result}")
    # 输出: [1, 1, 4, 2, 1, 1, 0, 0]
```

## 🔍 总结与应用场景

### 数据结构选择指南

| 数据结构 | 适用场景 | 时间复杂度 | 空间复杂度 |
|----------|----------|------------|------------|
| 树状数组 | 单点修改、区间查询 | O(log n) | O(n) |
| 并查集 | 动态连通性问题 | O(α(n)) | O(n) |
| 单调栈 | 寻找下一个更大/小元素 | O(n) | O(n) |
