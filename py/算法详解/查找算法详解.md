# 查找算法详解

## 📖 概述

查找算法是计算机科学中的基础算法，用于在数据集合中寻找特定元素。不同的查找算法适用于不同的数据结构和应用场景，选择合适的查找算法能够显著提高程序的效率。

---

## 🔍 线性查找

### 基本概念

线性查找（Linear Search）是最简单的查找算法，通过逐个检查数据集合中的每个元素来寻找目标值。

### 算法特点
- 时间复杂度：O(n)
- 空间复杂度：O(1)
- 适用于无序数据
- 实现简单

### 代码实现

```python
def linear_search(arr, target):
    """
    线性查找算法
    Args:
        arr: 待查找的数组
        target: 目标值
    Returns:
        目标值的索引，如果不存在返回-1
    """
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

def linear_search_all(arr, target):
    """
    查找所有匹配的元素位置
    """
    positions = []
    for i in range(len(arr)):
        if arr[i] == target:
            positions.append(i)
    return positions

# 测试用例
def test_linear_search():
    arr = [64, 34, 25, 12, 22, 11, 90, 25]
    target = 25
    
    result = linear_search(arr, target)
    all_results = linear_search_all(arr, target)
    
    print(f"数组: {arr}")
    print(f"查找目标: {target}")
    print(f"第一个位置: {result}")
    print(f"所有位置: {all_results}")
```

## 🎯 二分查找

### 基本概念

二分查找（Binary Search）是一种高效的查找算法，适用于已排序的数组。通过不断将查找区间对半分割来定位目标元素。

### 算法思路
1. 确定查找区间的中点
2. 比较中点值与目标值
3. 根据比较结果缩小查找区间
4. 重复直到找到目标或区间为空

### 基本实现

```python
def binary_search(arr, target):
    """
    二分查找 - 迭代实现
    时间复杂度: O(log n)
    空间复杂度: O(1)
    """
    left, right = 0, len(arr) - 1
    
    while left <= right:
        mid = left + (right - left) // 2  # 避免溢出
        
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return -1

def binary_search_recursive(arr, target, left=0, right=None):
    """
    二分查找 - 递归实现
    """
    if right is None:
        right = len(arr) - 1
    
    if left > right:
        return -1
    
    mid = left + (right - left) // 2
    
    if arr[mid] == target:
        return mid
    elif arr[mid] < target:
        return binary_search_recursive(arr, target, mid + 1, right)
    else:
        return binary_search_recursive(arr, target, left, mid - 1)

# 查找边界的变种
def binary_search_left_bound(arr, target):
    """
    查找目标值的左边界（第一个出现的位置）
    """
    left, right = 0, len(arr)
    
    while left < right:
        mid = left + (right - left) // 2
        
        if arr[mid] < target:
            left = mid + 1
        else:
            right = mid
    
    return left if left < len(arr) and arr[left] == target else -1

def binary_search_right_bound(arr, target):
    """
    查找目标值的右边界（最后一个出现的位置）
    """
    left, right = 0, len(arr)
    
    while left < right:
        mid = left + (right - left) // 2
        
        if arr[mid] <= target:
            left = mid + 1
        else:
            right = mid
    
    return left - 1 if left > 0 and arr[left - 1] == target else -1

# 测试用例
def test_binary_search():
    arr = [1, 2, 3, 4, 5, 5, 5, 6, 7, 8, 9]
    target = 5
    
    print(f"数组: {arr}")
    print(f"查找目标: {target}")
    print(f"基本二分查找: {binary_search(arr, target)}")
    print(f"左边界: {binary_search_left_bound(arr, target)}")
    print(f"右边界: {binary_search_right_bound(arr, target)}")
```

### 二分查找的应用

#### 1. 在旋转排序数组中查找

```python
def search_rotated_array(nums, target):
    """
    在旋转排序数组中查找目标值
    """
    left, right = 0, len(nums) - 1
    
    while left <= right:
        mid = left + (right - left) // 2
        
        if nums[mid] == target:
            return mid
        
        # 判断哪一半是有序的
        if nums[left] <= nums[mid]:  # 左半部分有序
            if nums[left] <= target < nums[mid]:
                right = mid - 1
            else:
                left = mid + 1
        else:  # 右半部分有序
            if nums[mid] < target <= nums[right]:
                left = mid + 1
            else:
                right = mid - 1
    
    return -1

# 测试用例
def test_rotated_search():
    nums = [4, 5, 6, 7, 0, 1, 2]
    target = 0
    result = search_rotated_array(nums, target)
    print(f"旋转数组: {nums}")
    print(f"查找 {target}: 位置 {result}")
```

#### 2. 寻找峰值

```python
def find_peak_element(nums):
    """
    寻找峰值元素（比相邻元素都大的元素）
    """
    left, right = 0, len(nums) - 1
    
    while left < right:
        mid = left + (right - left) // 2
        
        if nums[mid] > nums[mid + 1]:
            # 峰值在左半部分（包括mid）
            right = mid
        else:
            # 峰值在右半部分
            left = mid + 1
    
    return left

# 测试用例
def test_find_peak():
    nums = [1, 2, 3, 1]
    peak_idx = find_peak_element(nums)
    print(f"数组: {nums}")
    print(f"峰值位置: {peak_idx}, 峰值: {nums[peak_idx]}")
```

## 🔢 矩阵二分查找

### 基本概念

在有序矩阵中进行查找，可以利用矩阵的有序性质来优化查找过程。

### 算法思路
- 从矩阵的右上角或左下角开始
- 根据当前元素与目标值的比较结果决定移动方向
- 时间复杂度：O(m + n)

### 代码实现

```python
def search_matrix(matrix, target):
    """
    搜索二维矩阵（每行和每列都按升序排列）
    从右上角开始搜索
    """
    if not matrix or not matrix[0]:
        return False
    
    rows, cols = len(matrix), len(matrix[0])
    row, col = 0, cols - 1  # 从右上角开始
    
    while row < rows and col >= 0:
        current = matrix[row][col]
        
        if current == target:
            return True
        elif current > target:
            col -= 1  # 当前值太大，向左移动
        else:
            row += 1  # 当前值太小，向下移动
    
    return False

def search_matrix_sorted(matrix, target):
    """
    搜索二维矩阵（每行第一个元素大于前一行最后一个元素）
    可以看作一维有序数组进行二分查找
    """
    if not matrix or not matrix[0]:
        return False
    
    rows, cols = len(matrix), len(matrix[0])
    left, right = 0, rows * cols - 1
    
    while left <= right:
        mid = left + (right - left) // 2
        mid_value = matrix[mid // cols][mid % cols]
        
        if mid_value == target:
            return True
        elif mid_value < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return False

def search_matrix_with_position(matrix, target):
    """
    返回目标值在矩阵中的位置
    """
    if not matrix or not matrix[0]:
        return (-1, -1)
    
    rows, cols = len(matrix), len(matrix[0])
    row, col = 0, cols - 1
    
    while row < rows and col >= 0:
        current = matrix[row][col]
        
        if current == target:
            return (row, col)
        elif current > target:
            col -= 1
        else:
            row += 1
    
    return (-1, -1)

# 测试用例
def test_matrix_search():
    matrix1 = [
        [1,  4,  7,  11],
        [2,  5,  8,  12],
        [3,  6,  9,  16],
        [10, 13, 14, 17]
    ]
    
    matrix2 = [
        [1,  3,  5,  7],
        [10, 11, 16, 20],
        [23, 30, 34, 60]
    ]
    
    target = 5
    
    print("矩阵1 (每行每列有序):")
    for row in matrix1:
        print(row)
    print(f"查找 {target}: {search_matrix(matrix1, target)}")
    print(f"位置: {search_matrix_with_position(matrix1, target)}")
    
    print("\n矩阵2 (整体有序):")
    for row in matrix2:
        print(row)
    print(f"查找 {target}: {search_matrix_sorted(matrix2, target)}")
```

## 🔎 K最近邻查找算法

### 基本概念

K最近邻（K-Nearest Neighbors, KNN）算法用于在数据集中找到与查询点最相似的K个数据点。

### 算法思路
1. 计算查询点与所有数据点的距离
2. 按距离排序
3. 选择前K个最近的点

### 代码实现

```python
import math
import heapq
from collections import Counter

class Point:
    """表示多维空间中的点"""
    def __init__(self, coordinates, label=None):
        self.coordinates = coordinates
        self.label = label
        self.dimension = len(coordinates)
    
    def distance_to(self, other):
        """计算到另一个点的欧几里得距离"""
        if self.dimension != other.dimension:
            raise ValueError("维度不匹配")
        
        return math.sqrt(sum((a - b) ** 2 for a, b in 
                           zip(self.coordinates, other.coordinates)))
    
    def __repr__(self):
        return f"Point({self.coordinates}, {self.label})"

class KNNClassifier:
    """K最近邻分类器"""
    
    def __init__(self, k=3):
        self.k = k
        self.training_data = []
    
    def fit(self, points):
        """训练模型（存储训练数据）"""
        self.training_data = points
    
    def predict(self, query_point):
        """预测查询点的类别"""
        if len(self.training_data) < self.k:
            raise ValueError("训练数据不足")
        
        # 计算所有距离
        distances = []
        for point in self.training_data:
            dist = query_point.distance_to(point)
            distances.append((dist, point.label))
        
        # 找到K个最近的邻居
        distances.sort()
        k_nearest = distances[:self.k]
        
        # 投票决定类别
        labels = [label for _, label in k_nearest]
        return Counter(labels).most_common(1)[0][0]
    
    def find_k_nearest(self, query_point):
        """找到K个最近的邻居"""
        distances = []
        for point in self.training_data:
            dist = query_point.distance_to(point)
            distances.append((dist, point))
        
        distances.sort()
        return [(dist, point) for dist, point in distances[:self.k]]

# 优化版本：使用堆
class KNNOptimized:
    """优化的KNN实现，使用最大堆"""
    
    def __init__(self, k=3):
        self.k = k
        self.training_data = []
    
    def fit(self, points):
        self.training_data = points
    
    def find_k_nearest_heap(self, query_point):
        """使用堆优化的K最近邻查找"""
        max_heap = []  # 最大堆，存储(-distance, point)
        
        for point in self.training_data:
            dist = query_point.distance_to(point)
            
            if len(max_heap) < self.k:
                heapq.heappush(max_heap, (-dist, point))
            elif dist < -max_heap[0][0]:
                heapq.heapreplace(max_heap, (-dist, point))
        
        # 转换为正距离并排序
        result = [(-dist, point) for dist, point in max_heap]
        result.sort()
        return result
    
    def predict(self, query_point):
        """预测类别"""
        k_nearest = self.find_k_nearest_heap(query_point)
        labels = [point.label for _, point in k_nearest]
        return Counter(labels).most_common(1)[0][0]

# 测试用例
def test_knn():
    # 创建训练数据
    training_points = [
        Point([1, 1], 'A'),
        Point([2, 2], 'A'),
        Point([3, 3], 'A'),
        Point([6, 6], 'B'),
        Point([7, 7], 'B'),
        Point([8, 8], 'B'),
        Point([2, 6], 'C'),
        Point([3, 7], 'C'),
        Point([4, 8], 'C')
    ]
    
    # 创建分类器
    knn = KNNClassifier(k=3)
    knn.fit(training_points)
    
    # 测试查询点
    query_point = Point([4, 4])
    
    # 找到最近邻
    nearest = knn.find_k_nearest(query_point)
    print(f"查询点: {query_point}")
    print("3个最近邻:")
    for dist, point in nearest:
        print(f"  距离: {dist:.2f}, 点: {point}")
    
    # 预测类别
    prediction = knn.predict(query_point)
    print(f"预测类别: {prediction}")
    
    # 测试优化版本
    knn_opt = KNNOptimized(k=3)
    knn_opt.fit(training_points)
    
    nearest_opt = knn_opt.find_k_nearest_heap(query_point)
    prediction_opt = knn_opt.predict(query_point)
    
    print(f"\n优化版本预测: {prediction_opt}")
```

### KNN的应用场景

#### 1. 推荐系统

```python
class RecommendationSystem:
    """基于KNN的推荐系统"""
    
    def __init__(self, k=5):
        self.k = k
        self.users = {}  # 用户ID -> 评分向量
        self.items = set()
    
    def add_user_ratings(self, user_id, ratings):
        """添加用户评分数据"""
        self.users[user_id] = ratings
        self.items.update(ratings.keys())
    
    def cosine_similarity(self, ratings1, ratings2):
        """计算余弦相似度"""
        common_items = set(ratings1.keys()) & set(ratings2.keys())
        
        if not common_items:
            return 0
        
        sum1 = sum(ratings1[item] ** 2 for item in common_items)
        sum2 = sum(ratings2[item] ** 2 for item in common_items)
        sum_product = sum(ratings1[item] * ratings2[item] for item in common_items)
        
        denominator = math.sqrt(sum1 * sum2)
        return sum_product / denominator if denominator != 0 else 0
    
    def find_similar_users(self, target_user):
        """找到相似用户"""
        if target_user not in self.users:
            return []
        
        similarities = []
        target_ratings = self.users[target_user]
        
        for user_id, ratings in self.users.items():
            if user_id != target_user:
                similarity = self.cosine_similarity(target_ratings, ratings)
                similarities.append((similarity, user_id))
        
        similarities.sort(reverse=True)
        return similarities[:self.k]
    
    def recommend_items(self, target_user):
        """为用户推荐物品"""
        similar_users = self.find_similar_users(target_user)
        target_ratings = self.users[target_user]
        
        recommendations = {}
        
        for similarity, user_id in similar_users:
            user_ratings = self.users[user_id]
            
            for item, rating in user_ratings.items():
                if item not in target_ratings:  # 用户未评分的物品
                    if item not in recommendations:
                        recommendations[item] = 0
                    recommendations[item] += similarity * rating
        
        # 按推荐分数排序
        sorted_recommendations = sorted(recommendations.items(), 
                                      key=lambda x: x[1], reverse=True)
        
        return sorted_recommendations

# 测试推荐系统
def test_recommendation():
    rec_sys = RecommendationSystem(k=2)
    
    # 添加用户评分数据
    rec_sys.add_user_ratings('Alice', {'Movie1': 5, 'Movie2': 3, 'Movie3': 4})
    rec_sys.add_user_ratings('Bob', {'Movie1': 4, 'Movie2': 4, 'Movie4': 5})
    rec_sys.add_user_ratings('Charlie', {'Movie2': 2, 'Movie3': 5, 'Movie4': 4})
    rec_sys.add_user_ratings('David', {'Movie1': 5, 'Movie3': 4, 'Movie5': 3})
    
    # 为Alice推荐电影
    similar_users = rec_sys.find_similar_users('Alice')
    recommendations = rec_sys.recommend_items('Alice')
    
    print("Alice的相似用户:")
    for similarity, user in similar_users:
        print(f"  {user}: 相似度 {similarity:.3f}")
    
    print("\n为Alice推荐的电影:")
    for item, score in recommendations:
        print(f"  {item}: 推荐分数 {score:.3f}")
```

## 📊 查找算法总结

### 算法复杂度对比

| 算法 | 时间复杂度 | 空间复杂度 | 数据要求 | 适用场景 |
|------|------------|------------|----------|----------|
| 线性查找 | O(n) | O(1) | 无 | 小数据集、无序数据 |
| 二分查找 | O(log n) | O(1) | 有序 | 大数据集、静态数据 |
| 矩阵查找 | O(m+n) | O(1) | 行列有序 | 二维有序数据 |
| KNN | O(n) | O(n) | 无 | 分类、推荐系统 |

### 💡 学习建议

1. **理解适用场景**: 不同算法适用于不同的数据特征和应用需求
2. **掌握变种**: 熟练掌握各种算法的变种和优化方法
3. **实际应用**: 结合具体问题练习算法的应用
4. **性能分析**: 学会分析算法的时间和空间复杂度
5. **组合使用**: 在实际项目中灵活组合多种查找策略

### 🎯 选择指南

- **数据量小**: 使用线性查找
- **数据有序**: 优先考虑二分查找
- **频繁查找**: 考虑建立索引或哈希表
- **相似性查找**: 使用KNN或其他相似度算法
- **实时性要求高**: 选择时间复杂度低的算法
